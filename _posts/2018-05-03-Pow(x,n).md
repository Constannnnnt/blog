Implement pow(x, n), which calculates x raised to the power n (xn).

<!--excerpt-->

For example:

```
Input: 2.00000, 10
Output: 1024.00000
```

```
Input: 2.10000, 3
Output: 9.26100
```

```
Input: 2.00000, -2
Output: 0.25000
Explanation: 2-2 = 1/22 = 1/4 = 0.25
```

Solution:

```C++
class Solution {
public:
    double myPow(double x, int n) {
        if (n == INT_MIN && abs(x) != 1) {
            return 0;
        }
        if (n == INT_MIN && abs(x) == 1) {
            return 1;
        }
        if (n == 0) return 1;

        if (n < 0) {
            n = -n;
            x = 1 / x;
        }
        return (n % 2 == 0) ? myPow(x*x, n / 2) : x*myPow(x*x, n / 2);
    }
};
```
